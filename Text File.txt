    <script>
        // Init Lucide
        lucide.createIcons();

        // Standard Tab Routing Logic
        function switchTab(tabId) {
            // Hide all
            document.querySelectorAll('.tab-content').forEach(el => {
                el.classList.add('hidden');
                el.classList.remove('block');
            });
            document.querySelectorAll('.sidebar-item').forEach(el => el.classList.remove('active'));
            
            // Show target
            document.getElementById(`tab-${tabId}`).classList.remove('hidden');
            document.getElementById(`tab-${tabId}`).classList.add('block');
            document.getElementById(`nav-${tabId}`).classList.add('active');
        }

        // Custom non-blocking Toast feedback
        function saveRules() {
            const btn = event.currentTarget;
            const origHTML = btn.innerHTML;
            
            // UI Loading Simulation
            btn.innerHTML = `<div class="w-4 h-4 border-2 border-white border-t-transparent rounded-full animate-spin"></div> Saving...`;
            
            setTimeout(() => {
                btn.innerHTML = origHTML;
                
                // Show Toast
                const toast = document.getElementById('toast');
                toast.className = "fixed bottom-5 right-5 bg-green-50 text-green-800 border border-green-200 px-4 py-3 rounded-md shadow-lg flex items-center gap-2 transform translate-y-0 opacity-100 transition-all duration-300 z-50 text-sm font-medium";
                
                setTimeout(() => {
                    toast.className = "fixed bottom-5 right-5 bg-green-50 text-green-800 border border-green-200 px-4 py-3 rounded-md shadow-lg flex items-center gap-2 transform translate-y-20 opacity-0 transition-all duration-300 z-50 text-sm font-medium";
                }, 3000);
            }, 600);
        }
    </script>

            } else {

        // Logic to Mark a single row as Done
        function markRowDone(btnElement) {
            const row = btnElement.closest('.notif-row');

            showToast();

            setTimeout(() => {
                const group = row.closest('.notif-group');
                const remainingRows = group.querySelectorAll('.notif-row:not(.done)');
                
                // Only hide group if ALL remaining rows are truly done/hidden
                let visibleRemaining = 0;
                remainingRows.forEach(r => {
                    if (r.style.display !== 'none') visibleRemaining++;
                });

                if (remainingRows.length === 0 || visibleRemaining === 0) {
                    group.style.display = 'none'; 
                }
                
                checkEmptyState();
            }, 300);
        }

            if(activeCount === 0) return;

                    row.classList.add('done');
                    activeCount--;
                }
            });
            
            updateCounts();
            showToast("Visible notifications marked as done.");

                const groups = document.querySelectorAll('.notif-group');
                groups.forEach(group => {

                    if (remainingRows.length === 0) group.style.display = 'none';

                checkEmptyState();
